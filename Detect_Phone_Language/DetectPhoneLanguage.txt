' تابع تشخیص زبان گوشی
Sub GetLanguage As String
	Dim r As Reflector
	r.Target = r.RunStaticMethod("java.util.Locale", "getDefault", Null, Null)
	Return r.RunMethod("getDisplayName")
End Sub

	log(GetLanguage)
